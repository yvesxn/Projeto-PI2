function trocarImagem(imagem){ const principal = document.getElementById("imagemPrincipal"); principal.style.opacity = "0"; setTimeout(() => { principal.src = imagem; principal.style.opacity = "1"; }, 200); } const menuToggle = document.getElementById('menuToggle'); const nav = document.querySelector('nav'); if (menuToggle && nav) { menuToggle.addEventListener('click', () => { nav.classList.toggle('open'); }); nav.querySelectorAll('a').forEach(link => { link.addEventListener('click', () => { nav.classList.remove('open'); }); }); }
